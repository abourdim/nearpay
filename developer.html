Developer Stack

Full self-hosted NearPay stack. An ESP32-C3 reads NFC tags and POSTs to a Python server. The server broadcasts via WebSocket to all connected browsers in real time.

⚡
Stack summary: ESP32-C3 + PN532 NFC → WiFi → Python FastAPI → WebSocket → HTML/CSS/JS PWA. No cloud required. Runs on a Raspberry Pi or any Linux server.

# NearPay ESP32-C3 — main.py
# Reads NFC tags via PN532 I2C and POSTs to Python server

import network
import urequests
import ujson
import utime
from machine import I2C, Pin, reset
from pn532_i2c import PN532_I2C
from config import WIFI_SSID, WIFI_PASS, SERVER_URL, TABLE_ID

# ── LED for visual feedback ──
led = Pin(8, Pin.OUT)  # Built-in LED on some SuperMini boards

# ── Connect to WiFi ──
def connect_wifi():
    wlan = network.WLAN(network.STA_IF)
    wlan.active(True)
    if not wlan.isconnected():
        print(f"Connecting to {WIFI_SSID}...")
        wlan.connect(WIFI_SSID, WIFI_PASS)
        timeout = 15
        while not wlan.isconnected() and timeout > 0:
            utime.sleep(1)
            timeout -= 1
    if wlan.isconnected():
        print(f"✅ WiFi OK — IP: {wlan.ifconfig()[0]}")
        return True
    print("❌ WiFi failed — rebooting in 5s")
    utime.sleep(5)
    reset()

# ── POST tap event to Python server ──
def notify_server(uid_hex: str, raw_uid: list):
    payload = {
        "table_id":  TABLE_ID,
        "uid":       uid_hex,
        "uid_bytes": raw_uid,
        "ts":        utime.time(),
        "rssi":      network.WLAN(network.STA_IF).status("rssi")
    }
    try:
        r = urequests.post(
            SERVER_URL + "/nfc/tap",
            data=ujson.dumps(payload),
            headers={"Content-Type": "application/json"}
        )
        resp = r.json()
        r.close()
        return resp
    except Exception as e:
        print(f"❌ Server error: {e}")
        return None

# ── LED blink feedback ──
def blink(times=1, delay=0.1):
    for _ in range(times):
        led.on(); utime.sleep(delay)
        led.off(); utime.sleep(delay)

# ── Main loop ──
def main():
    connect_wifi()

    # Init PN532 on I2C (SDA=GPIO8, SCL=GPIO9)
    i2c = I2C(0, sda=Pin(8), scl=Pin(9), freq=100000)
    nfc = PN532_I2C(i2c, debug=False, reset=Pin(10))
    nfc.SAM_configuration()  # Normal mode, no timeout

    fw = nfc.firmware_version()
    print(f"✅ PN532 ready — FW: {fw[1]}.{fw[2]}")
    print(f"📡 Listening on TABLE_ID: {TABLE_ID}")

    last_uid = None
    last_tap_time = 0

    while True:
        uid = nfc.read_passive_target(timeout=500)

        if uid is not None:
            uid_hex = "".join([f"{b:02X}" for b in uid])
            now = utime.ticks_ms()

            # Debounce: ignore same tag within 3 seconds
            if uid_hex != last_uid or utime.ticks_diff(now, last_tap_time) > 3000:
                print(f"\n🏷️ Tag detected: {uid_hex}")
                blink(2)  # Visual confirmation

                resp = notify_server(uid_hex, list(uid))

                if resp:
                    print(f"✅ Server: {resp.get('status')} — {resp.get('table')}")
                    blink(3, 0.05)  # Success blink
                else:
                    print("⚠️ No server response")

                last_uid = uid_hex
                last_tap_time = now

        utime.sleep_ms(100)  # Poll every 100ms

if __name__ == "__main__":
    main()
📄 pn532_i2c.py MicroPython
# PN532 I2C driver for MicroPython
# Supports passive NFC target reading (ISO 14443A)

import utime
from machine import I2C, Pin

PN532_I2C_ADDRESS  = 0x24
PN532_PREAMBLE     = 0x00
PN532_STARTCODE1   = 0x00
PN532_STARTCODE2   = 0xFF
PN532_HOSTTOPN532  = 0xD4
PN532_PN532TOHOST  = 0xD5
PN532_ACK_WAIT_TIME = 10  # ms

# Command codes
CMD_SAMCONFIGURATION    = 0x14
CMD_GETFIRMWAREVERSION  = 0x02
CMD_INLISTPASSIVETARGET = 0x4A

class PN532_I2C:
    def __init__(self, i2c: I2C, reset=None, debug=False):
        self.i2c   = i2c
        self.debug = debug
        self._reset_pin = reset
        if reset:
            reset.init(Pin.OUT, value=1)
            utime.sleep_ms(100)
            reset.value(0); utime.sleep_ms(400)
            reset.value(1); utime.sleep_ms(10)
        self._wakeup()

    def _wakeup(self):
        # Send I2C wakeup sequence
        try:
            self.i2c.writeto(PN532_I2C_ADDRESS, bytes([0x55]))
        except:
            pass
        utime.sleep_ms(500)

    def _write_frame(self, data: bytes):
        length = len(data)
        lcs = (~length + 1) & 0xFF
        dcs = (~sum(data) + 1) & 0xFF
        frame = bytes([
            PN532_PREAMBLE, PN532_STARTCODE1, PN532_STARTCODE2,
            length, lcs, PN532_HOSTTOPN532
        ]) + data + bytes([dcs, 0x00])
        self.i2c.writeto(PN532_I2C_ADDRESS, frame)

    def _read_data(self, count: int) -> bytes:
        # Poll until ready (first byte = 0x01)
        for _ in range(20):
            data = self.i2c.readfrom(PN532_I2C_ADDRESS, count + 1)
            if data[0] == 0x01:
                return data[1:]
            utime.sleep_ms(PN532_ACK_WAIT_TIME)
        raise RuntimeError("PN532 timeout")

    def _call_function(self, cmd: int, params=None, response_len=0):
        data = bytes([cmd]) + (bytes(params) if params else b"")
        self._write_frame(data)
        utime.sleep_ms(PN532_ACK_WAIT_TIME)
        resp = self._read_data(response_len + 8)
        return resp[7:7+response_len] if response_len > 0 else None

    def firmware_version(self):
        resp = self._call_function(CMD_GETFIRMWAREVERSION, response_len=4)
        return resp  # (IC, Ver, Rev, Support)

    def SAM_configuration(self):
        # Normal mode, no timeout, no IRQ
        self._call_function(CMD_SAMCONFIGURATION, [0x01, 0x14, 0x01])

    def read_passive_target(self, card_baud=0x00, timeout=1000) -> bytes | None:
        # ISO 14443A (NTAG, MIFARE). Returns UID bytes or None.
        self._write_frame(bytes([CMD_INLISTPASSIVETARGET, 0x01, card_baud]))
        deadline = utime.ticks_ms() + timeout
        while utime.ticks_diff(deadline, utime.ticks_ms()) > 0:
            try:
                resp = self._read_data(19)
                if resp[6] == 1:  # Target found
                    uid_len = resp[12]
                    return bytes(resp[13:13+uid_len])
            except:
                pass
            utime.sleep_ms(50)
        return None

from fastapi import FastAPI, WebSocket, WebSocketDisconnect, HTTPException
from fastapi.staticfiles import StaticFiles
from fastapi.middleware.cors import CORSMiddleware
from fastapi.responses import FileResponse
from contextlib import asynccontextmanager
import asyncio, ujson as json, sqlite3
from datetime import datetime
from models import NFCTapEvent, OrderItem
from pos_mock import get_table_order, close_order

# ── WebSocket connection manager ──
class ConnectionManager:
    def __init__(self):
        self.active: dict[str, list[WebSocket]] = {}  # room → clients

    async def connect(self, ws: WebSocket, room: str):
        await ws.accept()
        self.active.setdefault(room, []).append(ws)
        print(f"🔌 WS connected — room={room} total={len(self.active[room])}")

    def disconnect(self, ws: WebSocket, room: str):
        if room in self.active:
            self.active[room].discard(ws)

    async def broadcast(self, room: str, message: dict):
        if room not in self.active: return
        dead = []
        for ws in self.active[room]:
            try:
                await ws.send_json(message)
            except:
                dead.append(ws)
        for ws in dead:
            self.active[room].discard(ws)

manager = ConnectionManager()

# ── Database setup ──
def init_db():
    db = sqlite3.connect("nearpay.db")
    db.execute("""
        CREATE TABLE IF NOT EXISTS taps (
            id       INTEGER PRIMARY KEY AUTOINCREMENT,
            table_id TEXT NOT NULL,
            uid      TEXT NOT NULL,
            ts       INTEGER NOT NULL,
            venue_id TEXT,
            status   TEXT DEFAULT 'pending'
        )""")
    db.execute("""
        CREATE TABLE IF NOT EXISTS payments (
            id         INTEGER PRIMARY KEY AUTOINCREMENT,
            table_id   TEXT NOT NULL,
            amount     INTEGER NOT NULL,
            currency   TEXT DEFAULT 'EUR',
            status     TEXT DEFAULT 'pending',
            method     TEXT,
            created_at TEXT,
            paid_at    TEXT
        )""")
    db.commit()
    db.close()

# ── App setup ──
@asynccontextmanager
async def lifespan(app: FastAPI):
    init_db()
    print("✅ NearPay server started")
    yield

app = FastAPI(title="NearPay API", version="1.0.0", lifespan=lifespan)
app.add_middleware(CORSMiddleware, allow_origins=["*"], allow_methods=["*"], allow_headers=["*"])
app.mount("/static", StaticFiles(directory="static"), name="static")

# ── ESP32 NFC tap endpoint ──
@app.post("/nfc/tap")
async def nfc_tap(event: NFCTapEvent):
    print(f"📡 NFC tap — table={event.table_id} uid={event.uid}")

    # Save tap to DB
    db = sqlite3.connect("nearpay.db")
    db.execute(
        "INSERT INTO taps (table_id, uid, ts) VALUES (?, ?, ?)",
        (event.table_id, event.uid, event.ts)
    )
    db.commit(); db.close()

    # Fetch order from mock POS
    order = get_table_order(event.table_id)

    # Broadcast to all WebSocket clients in this venue
    await manager.broadcast(event.table_id, {
        "event": "nfc_tap",
        "table": event.table_id,
        "uid":   event.uid,
        "order": order,
        "ts":    event.ts,
        "url":   f"http://server:8000/t/{event.table_id}"
    })

    return {"status": "ok", "table": event.table_id, "order": order}

# ── Process payment ──
@app.post("/api/pay/{table_id}")
async def process_payment(table_id: str, data: dict):
    order = get_table_order(table_id)
    if not order:
        raise HTTPException(404)

    # Save payment
    db = sqlite3.connect("nearpay.db")
    db.execute(
        "INSERT INTO payments (table_id, amount, method, status, created_at, paid_at) VALUES (?,?,?,?,?,?)",
        (table_id, order["total"], data.get("method", "card"), "paid",
         datetime.now().isoformat(), datetime.now().isoformat())
    )
    db.commit(); db.close()

    # Close order in POS
    close_order(table_id)

    # Broadcast payment success
    await manager.broadcast(table_id, {
        "event": "payment_success",
        "table": table_id,
        "amount": order["total"],
        "ts": datetime.now().isoformat()
    })
    # Also broadcast to dashboard
    await manager.broadcast("dashboard", {
        "event": "table_paid",
        "table": table_id,
        "amount": order["total"]
    })
    # Broadcast to kitchen
    await manager.broadcast("kitchen", {
        "event": "table_paid",
        "table": table_id
    })

    return {"status": "paid", "receipt_id": f"NP-{table_id}-{int(datetime.now().timestamp())}"}

from pydantic import BaseModel
from typing import Optional, List
from datetime import datetime

class NFCTapEvent(BaseModel):
    table_id:  str
    uid:       str           # NFC UID hex string e.g. "A3F9B2C1"
    uid_bytes: List[int]     # Raw bytes [163, 249, 178, 193]
    ts:        int           # Unix timestamp from ESP32
    rssi:      Optional[int] # WiFi signal strength (dBm)
    venue_id:  Optional[str]

class OrderItem(BaseModel):
    sku:         str
    name:        str
    qty:         int
    unit_price:  int         # in cents
    line_total:  int

class Order(BaseModel):
    table_id:   str
    order_id:   str
    status:     str          # "open" | "paying" | "closed"
    items:      List[OrderItem]
    subtotal:   int
    tax:        int
    total:      int          # in cents
    currency:   str = "EUR"
    opened_at:  datetime
    server_id:  Optional[str]

class PaymentRequest(BaseModel):
    table_id:  str
    amount:    int           # must match order total
    method:    str           # "apple_pay" | "google_pay" | "card"
    token:     Optional[str] # Stripe payment method token
    split:     Optional[int] # number of people splitting

class WebSocketMessage(BaseModel):
    event:  str
    table:  Optional[str]
    data:   Optional[dict]
    ts:     Optional[str]

<script>
  const ws = new WebSocket(`ws://${location.hostname}:8000/ws/kitchen`);
  const orders = new Map();

  ws.onmessage = (e) => {
    const msg = JSON.parse(e.data);

    switch(msg.event) {
      case 'new_order_item':
        // New item ordered from customer phone
        addKDSCard({
          table: msg.table,
          item: msg.item,
          qty: msg.qty,
          note: msg.note,
          ts: Date.now()
        });
        playChime();
        break;

      case 'table_paid':
        // Mark table cards as complete
        markTableDone(msg.table);
        break;
    }
  };

  function addKDSCard(order) {
    const card = document.createElement('div');
    card.className = 'kds-card';
    card.dataset.table = order.table;
    card.innerHTML = `
      <div class="kds-table">TABLE ${order.table.replace('TABLE-','')}</div>
      <div class="kds-item">${order.qty}× ${order.item}</div>
      ${order.note ? `<div class="kds-note">📝 ${order.note}</div>` : ''}
      <div class="kds-timer" id="timer-${Date.now()}">0:00</div>
    `;
    document.getElementById('kds-grid').appendChild(card);
    startTimer(card.querySelector('.kds-timer'));
  }

  function startTimer(el) {
    const start = Date.now();
    setInterval(() => {
      const s = Math.floor((Date.now()-start)/1000);
      const m = Math.floor(s/60);
      el.textContent = `${m}:${(s%60).toString().padStart(2,'0')}`;
      el.className = `kds-timer${s>300?' urgent':s>120?' warn':''}`;
    }, 1000);
  }

  function playChime() {
    const ctx = new AudioContext();
    const osc = ctx.createOscillator();
    osc.connect(ctx.destination);
    osc.frequency.value = 880;
    osc.start(); osc.stop(ctx.currentTime + 0.15);
  }
</script>

Setup Guide

Get Running
in 15 minutes

Flash MicroPython on ESP32-C3

Download MicroPython for ESP32-C3 from micropython.org. Flash with esptool.

Shell
pip install esptool thonny
esptool.py --chip esp32c3 erase_flash
esptool.py --chip esp32c3 write_flash 0x0 micropython-esp32c3.bin

Upload ESP32 files via Thonny

Open Thonny IDE. Connect to ESP32-C3. Upload main.py, pn532_i2c.py, config.py. Edit config.py with your WiFi and server IP.

Install Python server

Shell
git clone https://github.com/your-org/nearpay-server
cd nearpay-server
pip install -r requirements.txt
mkdir static
cp *.html static/     # Copy frontend files

Start the Python server

Shell
uvicorn server:app --host 0.0.0.0 --port 8000 --reload

# Output:
# ✅ NearPay server started
# INFO: Uvicorn running on http://0.0.0.0:8000

Wire PN532 to ESP32-C3 and power up

Connect 4 wires: GND→GND, 3.3V→VCC, GPIO8→SDA, GPIO9→SCL. Power the ESP32-C3 via USB. Watch the serial monitor — you should see the WiFi connection and PN532 initialization.

Write a table ID to an NFC tag

Python
# Run once to write URL to NTAG213 sticker
from machine import I2C, Pin
from pn532_i2c import PN532_I2C

i2c = I2C(0, sda=Pin(8), scl=Pin(9), freq=100000)
nfc = PN532_I2C(i2c)
nfc.SAM_configuration()

url = b"http://192.168.1.100:8000/t/TABLE-12"
# Write NDEF URI record to tag (page 4+)
nfc.ntag2xx_write_block(4, url[:4])
print("✅ Tag written!")

Open the dashboard and tap!

Open http://your-server-ip:8000/static/dashboard.html on your manager device. Place a phone on the NFC badge — the order should appear in under 1 second.

✅
Working! ESP32-C3 reads tag → POSTs to Python → Python broadcasts via WebSocket → Dashboard updates in real time → Customer phone opens payment page.
